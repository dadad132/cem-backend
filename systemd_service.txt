# CRM Backend Systemd Service Configuration
# This file shows how to set up the CRM as a system service on Ubuntu
#
# Installation instructions:
# 1. Copy this content to: /etc/systemd/system/crm-backend.service
# 2. Replace USER with your Ubuntu username
# 3. Replace /path/to/crm-backend with actual path
# 4. Run: sudo systemctl daemon-reload
# 5. Run: sudo systemctl enable crm-backend
# 6. Run: sudo systemctl start crm-backend
#
# Commands:
#   Start:   sudo systemctl start crm-backend
#   Stop:    sudo systemctl stop crm-backend
#   Restart: sudo systemctl restart crm-backend
#   Status:  sudo systemctl status crm-backend
#   Logs:    sudo journalctl -u crm-backend -f

[Unit]
Description=CRM Backend Server
After=network.target

[Service]
Type=simple
User=USER
WorkingDirectory=/path/to/crm-backend
Environment="PATH=/path/to/crm-backend/venv/bin"
ExecStart=/path/to/crm-backend/venv/bin/python /path/to/crm-backend/start_server.py
Restart=always
RestartSec=10

# Logging
StandardOutput=append:/path/to/crm-backend/logs/server.log
StandardError=append:/path/to/crm-backend/logs/error.log

[Install]
WantedBy=multi-user.target
